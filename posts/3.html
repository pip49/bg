GIF 번호 관리용 코드 (추천)

이 방식은 gifNumbers 안에 숫자만 넣으면 되어 아주 간편합니다

추가로 코드를 수정하시다가 특정 번호들(예: 1, 3, 5번)이나 특정 구간(예: 10~20번)을 GIF로 한 번에 관리하고 싶어지시면, 아래의 최종 템플릿을 복사해서 index.html에 붙여넣기만 하세요.

 

 

// 1. 초기 갤러리 생성 (49개)

for (let i = 1; i <= 49; i++) {

    const item = document.createElement('div');

    item.className = 'item';

 

    // --- 여기만 수정하면 됩니다 ---

    const gifNumbers = [3, 7, 8, 9]; // GIF로 표시할 번호들을 여기에 적으세요.

   

    // 조건: 번호가 gifNumbers에 포함되거나, 특정 구간(예: 20~25)일 때 gif 사용

    const isGif = gifNumbers.includes(i) || (i >= 20 && i <= 25);

   

    const imgPath = isGif ? `assets/images/${i}.gif` : `assets/images/${i}.png`;

    // ----------------------------

 

    const audioPath = `assets/audio/${i}.mp3`;

 

    item.innerHTML = `

        <img src="${imgPath}" alt="Amp ${i}" loading="lazy">

        <div class="item-title">AMPLIFIER #${i}</div>

    `;

 

    item.onclick = () => {

        startLoadingProcess(i, imgPath, audioPath);

    };

 

    gallery.appendChild(item);

}
